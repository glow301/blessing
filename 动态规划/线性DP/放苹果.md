### Code
##### C++
```cpp
#include<cstdio>

using namespace std;

int f(int n, int m) {
    if (n == 0 || m == 1) {
        return 1;
    }
    if (n < m) {
        return f(n, n);
    }
    return f(n, m-1) + f(n-m, m);
}

int main() {
    int T, n, m;
    scanf("%d", &T);
    while (T--) {
        scanf("%d %d", &n, &m);
        printf("%d\n", f(n, m));
    }
}
```

```go
package main

import "fmt"

const MAXN = 25
var dp [MAXN][MAXN]int

func f(n, m int) int {
    if n == 0 || m == 1 {
        return 1
    }
    if dp[n][m] != 0 {
        return dp[n][m]
    }
    if n < m {
        dp[n][n] = f(n, n);
        return dp[n][n]
    }
    dp[n][m-1] = f(n, m-1)
    dp[n-m][m] = f(n-m, m)
    return dp[n][m-1] + dp[n-m][m]
}

func main() {
    T, n, m := 0, 0, 0
    fmt.Scanf("%d\n", &T)
    for i := 0; i < T; i++ {
        fmt.Scanf("%d %d\n", &n, &m)
        fmt.Println(f(n, m))
    }
}
```
